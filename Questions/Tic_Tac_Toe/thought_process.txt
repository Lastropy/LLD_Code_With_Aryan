class PlayerFactory {
    static int id = 0;
    int getPlayer(PlayerType t){
        Symbol currentSymbol = Symbol.values()[id];
        if(t == PlayerType.HUMAN) {
            return Player(id++, currentSymbol, HumanPlayerStrategy);
        } else if(t == PlayerType.AI) {
            return Player(id++, currentSymbol, AIPlayerStrategy);
        } else {
            throw Exception("Invalid Player Type cannot be created");
        }
    }
}
